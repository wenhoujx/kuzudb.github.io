15 static inline void operation(common::ku_string_t& src, int64_t start, int64_t len,
16 common::ku_string_t& result, common::ValueVector& resultValueVector) {
17 std::string srcStr = src.getAsString();
18 bool isAscii = true;
19 auto startPos = start - 1;
20 auto endPos = std::min(srcStr.size(), (size_t)(startPos + len));
21 // 1 character more than length has to be scanned for diatrics case: y + ˘ = ў.
22 for (auto i = 0; i < std::min(srcStr.size(), endPos + 1); i++) {
23 // UTF-8 character encountered.
24 if (srcStr[i] & 0x80) {
25 isAscii = false;
26 break;
27 }
28 }
29 if (isAscii) {
30 copySubstr(src, start, len, result, resultValueVector, true /* isAscii */);
31 } else {
32 int64_t characterCount = 0, startBytePos = 0, endBytePos = 0;
33 kuzu::utf8proc::utf8proc_grapheme_callback(
34 srcStr.c_str(), srcStr.size(), [&](int64_t gstart, int64_t gend) {
35 if (characterCount == startPos) {
36 startBytePos = gstart;
37 } else if (characterCount == endPos) {
38 endBytePos = gstart;
39 return false;
40 }
41 characterCount++;
42 return true;
43 });
44 if (endBytePos == 0 && len != 0) {
45 endBytePos = src.len;
46 }
47 // In this case, the function gets the EXACT byte location to start copying from.
48 copySubstr(src, startBytePos, endBytePos - startBytePos, result, resultValueVector,
49 false /* isAscii */);
50 }
51 }

53 static inline void copySubstr(common::ku_string_t& src, int64_t start, int64_t len,
54 common::ku_string_t& result, common::ValueVector& resultValueVector, bool isAscii) {
55 result.len = std::min(len, src.len - start + 1);
56 if (!common::ku_string_t::isShortString(result.len)) {
57 result.overflowPtr = reinterpret_cast<uint64_t>(
58 resultValueVector.getOverflowBuffer().allocateSpace(result.len));
59 }
60 if (isAscii) {
61 // For normal ASCII char case, we get to the proper byte position to copy from by doing
62 // a -1 (since it is guaranteed each char is 1 byte).
63 memcpy((uint8_t*)result.getData(), src.getData() + start - 1, result.len);
64 } else {
65 // For utf8 char copy, the function gets the exact starting byte position to copy from.
66 memcpy((uint8_t*)result.getData(), src.getData() + start, result.len);
67 }
68 if (!common::ku_string_t::isShortString(result.len)) {
69 memcpy(result.prefix, result.getData(), common::ku_string_t::PREFIX_LENGTH);
70 }
71 }
