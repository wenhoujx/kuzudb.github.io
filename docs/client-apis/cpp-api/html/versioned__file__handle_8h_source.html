11// VersionedFileHandle is the in-memory representation of a file whose pages can have multiple
12// versions. It extends FileHandle with several data structures to keep track of the pageIdx's of
13// the WAL versions of the pages. This is intended to be used by storage structures that maintain
14// multiple versions of their pages (backed by WAL pages) and contains the helper functions they
15// commonly need.

22 // This function assumes that the caller has already acquired the lock for originalPageIdx.
23 inline bool hasWALPageVersionNoPageLock(common::page_idx_t originalPageIdx) {
24 auto pageGroupIdxAndPosInGroup = PageUtils::getPageElementCursorForPos(
25 originalPageIdx, common::MULTI_VERSION_FILE_PAGE_GROUP_SIZE);
26 // TODO: and Warning: This can be a major performance bottleneck, where we need to acquire a
27 // shared lock for the entire file handle to ensure that another thread cannot be updating
28 // pageVersions. We can fix this by instead making pageVersions a linkedList of vectors for
29 // each page group (so not a vector of vector, which can be dynamically allocated) and
30 // having a lock on each page group.
31 std::shared_lock slock(fhSharedMutex);
32 // There is an updated wal page if the PageVersionAndLockInfo for the page group exists
33 // and the page version for the page is not null (which is UINT32_MAX).
34 auto retVal = !pageVersions[pageGroupIdxAndPosInGroup.pageIdx].empty() &&
35 (pageVersions[pageGroupIdxAndPosInGroup.pageIdx]
36 [pageGroupIdxAndPosInGroup.elemPosInPage] != UINT32_MAX);
37 return retVal;
38 }

40 // This function assumes that the caller has already acquired the lock for originalPageIdx.
41 inline common::page_idx_t getWALPageVersionNoPageLock(common::page_idx_t originalPageIdx) {
42 // See the comment about a shared lock in hasWALPageVersionNoPageLock
43 std::shared_lock slock(fhSharedMutex);
44 auto pageGroupIdxAndPosInGroup = PageUtils::getPageElementCursorForPos(
45 originalPageIdx, common::MULTI_VERSION_FILE_PAGE_GROUP_SIZE);
46 return pageVersions[pageGroupIdxAndPosInGroup.pageIdx]
47 [pageGroupIdxAndPosInGroup.elemPosInPage];
48 }
