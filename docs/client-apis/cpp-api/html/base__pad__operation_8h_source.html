40 static std::pair<uint32_t, uint32_t> padCountChars(
41 const uint32_t count, const char* data, const uint32_t size) {
42 auto str = reinterpret_cast<const utf8proc::utf8proc_uint8_t*>(data);
43 uint32_t byteCount = 0, charCount = 0;
44 for (; charCount < count && byteCount < size; charCount++) {
45 utf8proc::utf8proc_int32_t codepoint;
46 auto bytes = utf8proc::utf8proc_iterate(str + byteCount, size - byteCount, &codepoint);
47 byteCount += bytes;
48 }
49 return {byteCount, charCount};
50 }

52 static void insertPadding(uint32_t charCount, common::ku_string_t pad, std::string& result) {
53 auto padData = pad.getData();
54 auto padSize = pad.len;
55 uint32_t padByteCount = 0;
56 for (auto i = 0; i < charCount; i++) {
57 if (padByteCount >= padSize) {
58 result.insert(result.end(), (char*)padData, (char*)(padData + padByteCount));
59 padByteCount = 0;
60 }
61 utf8proc::utf8proc_int32_t codepoint;
62 auto bytes = utf8proc::utf8proc_iterate(
63 padData + padByteCount, padSize - padByteCount, &codepoint);
64 padByteCount += bytes;
65 }
66 result.insert(result.end(), (char*)padData, (char*)(padData + padByteCount));
67 }
