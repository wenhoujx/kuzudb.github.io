199// StorageStructureIDs start with 1 byte type and 1 byte isOverflow field followed with additional
200// bytes needed by the different log types. We don't need these to be byte aligned because they are
201// not stored in memory. These are used to serialize and deserialize log entries.
202struct StorageStructureID {
203 StorageStructureType storageStructureType;
204 bool isOverflow;
205 union {
206 ColumnFileID columnFileID;
207 ListFileID listFileID;
208 NodeIndexID nodeIndexID;
209 };
210
211 inline bool operator==(const StorageStructureID& rhs) const {
212 if (storageStructureType != rhs.storageStructureType || isOverflow != rhs.isOverflow) {
213 return false;
214 }
215 switch (storageStructureType) {
216 case StorageStructureType::COLUMN: {
217 return columnFileID == rhs.columnFileID;
218 }
219 case StorageStructureType::LISTS: {
220 return listFileID == rhs.listFileID;
221 }
222 case StorageStructureType::NODE_INDEX: {
223 return nodeIndexID == rhs.nodeIndexID;
224 }
225 default: {
226 assert(false);
227 }
228 }
229 }

407struct WALRecord {
408 WALRecordType recordType;
409 union {
410 PageUpdateOrInsertRecord pageInsertOrUpdateRecord;
411 CommitRecord commitRecord;
412 NodeTableRecord nodeTableRecord;
413 RelTableRecord relTableRecord;
414 DiskOverflowFileNextBytePosRecord diskOverflowFileNextBytePosRecord;
415 CopyNodeRecord copyNodeRecord;
416 CopyRelRecord copyRelRecord;
417 TableStatisticsRecord tableStatisticsRecord;
418 DropTableRecord dropTableRecord;
419 DropPropertyRecord dropPropertyRecord;
420 AddPropertyRecord addPropertyRecord;
421 };
422
423 bool operator==(const WALRecord& rhs) const {
424 if (recordType != rhs.recordType) {
425 return false;
426 }
427 switch (recordType) {
428 case WALRecordType::PAGE_UPDATE_OR_INSERT_RECORD: {
429 return pageInsertOrUpdateRecord == rhs.pageInsertOrUpdateRecord;
430 }
431 case WALRecordType::COMMIT_RECORD: {
432 return commitRecord == rhs.commitRecord;
433 }
434 case WALRecordType::TABLE_STATISTICS_RECORD: {
435 return tableStatisticsRecord == rhs.tableStatisticsRecord;
436 }
437 case WALRecordType::CATALOG_RECORD: {
438 // CatalogRecords are empty so are always equal
439 return true;
440 }
441 case WALRecordType::NODE_TABLE_RECORD: {
442 return nodeTableRecord == rhs.nodeTableRecord;
443 }
444 case WALRecordType::REL_TABLE_RECORD: {
445 return relTableRecord == rhs.relTableRecord;
446 }
447 case WALRecordType::OVERFLOW_FILE_NEXT_BYTE_POS_RECORD: {
448 return diskOverflowFileNextBytePosRecord == rhs.diskOverflowFileNextBytePosRecord;
449 }
450 case WALRecordType::COPY_NODE_RECORD: {
451 return copyNodeRecord == rhs.copyNodeRecord;
452 }
453 case WALRecordType::COPY_REL_RECORD: {
454 return copyRelRecord == rhs.copyRelRecord;
455 }
456 case WALRecordType::DROP_TABLE_RECORD: {
457 return dropTableRecord == rhs.dropTableRecord;
458 }
459 case WALRecordType::DROP_PROPERTY_RECORD: {
460 return dropPropertyRecord == rhs.dropPropertyRecord;
461 }
462 case WALRecordType::ADD_PROPERTY_RECORD: {
463 return addPropertyRecord == rhs.addPropertyRecord;
464 }
465 default: {
466 throw common::RuntimeException("Unrecognized WAL record type inside ==. recordType: " +
467 walRecordTypeToString(recordType));
468 }
469 }
470 }
