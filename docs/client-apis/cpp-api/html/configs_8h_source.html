13// Currently the system supports files with 2 different pages size, which we refer to as
14// DEFAULT_PAGE_SIZE and LARGE_PAGE_SIZE. Default size of the page which is the unit of read/write
15// to the database files, such as to store columns or lists. For now, this value cannot be changed.
16// But technically it can change from 2^12 to 2^16. 2^12 lower bound is assuming the OS page size is
17// 4K. 2^16 is because currently we leave 11 fixed number of bits for relOffInPage and the maximum
18// number of bytes needed for an edge is 20 bytes so 11 + log_2(20) = 15.xxx, so certainly over
19// 2^16-size pages, we cannot utilize the page for storing adjacency lists.
20constexpr uint64_t DEFAULT_PAGE_SIZE_LOG_2 = 12;
21constexpr uint64_t DEFAULT_PAGE_SIZE = 1 << DEFAULT_PAGE_SIZE_LOG_2;
22// Page size for files with large pages, e.g., temporary files that are used by operators that may
23// require large amounts of memory.
24constexpr const uint64_t LARGE_PAGE_SIZE_LOG_2 = 18;
25constexpr uint64_t LARGE_PAGE_SIZE = 1 << LARGE_PAGE_SIZE_LOG_2;
