15// This class contains factorizedTables, nextFactorizedTableIdx, strKeyColsInfo,
16// sortedKeyBlocks and the size of each tuple in keyBlocks. The class is shared between the
17// order_by, orderByMerge, orderByScan operators. All functions are guaranteed to be thread-safe, so
18// caller doesn't need to acquire a lock before calling these functions.
19class SharedFactorizedTablesAndSortedKeyBlocks {
20public:
21 explicit SharedFactorizedTablesAndSortedKeyBlocks()
22 : nextFactorizedTableIdx{0},
23 sortedKeyBlocks{std::make_shared<std::queue<std::shared_ptr<MergedKeyBlocks>>>()} {}
24
25 uint8_t getNextFactorizedTableIdx() {
26 std::unique_lock lck{mtx};
27 return nextFactorizedTableIdx++;
28 }
29
30 void appendFactorizedTable(
31 uint8_t factorizedTableIdx, std::shared_ptr<FactorizedTable> factorizedTable) {
32 std::unique_lock lck{mtx};
33 // If the factorizedTables is full, resize the factorizedTables and
34 // insert the factorizedTable to the set.
35 if (factorizedTableIdx >= factorizedTables.size()) {
36 factorizedTables.resize(factorizedTableIdx + 1);
37 }
38 factorizedTables[factorizedTableIdx] = move(factorizedTable);
39 }
40
41 void appendSortedKeyBlock(std::shared_ptr<MergedKeyBlocks> mergedDataBlocks) {
42 std::unique_lock lck{mtx};
43 sortedKeyBlocks->emplace(mergedDataBlocks);
44 }
45
46 void setNumBytesPerTuple(uint32_t _numBytesPerTuple) {
47 assert(numBytesPerTuple == UINT32_MAX);
48 numBytesPerTuple = _numBytesPerTuple;
49 }
50
51 void combineFTHasNoNullGuarantee() {
52 for (auto i = 1u; i < factorizedTables.size(); i++) {
53 factorizedTables[0]->mergeMayContainNulls(*factorizedTables[i]);
54 }
55 }
56
57 void setStrKeyColInfo(std::vector<StrKeyColInfo> _strKeyColsInfo) {
58 assert(strKeyColsInfo.empty());
59 strKeyColsInfo = std::move(_strKeyColsInfo);
60 }
61
62private:
63 std::mutex mtx;
64
65public:
66 std::vector<std::shared_ptr<FactorizedTable>> factorizedTables;
67 uint8_t nextFactorizedTableIdx;
68 std::shared_ptr<std::queue<std::shared_ptr<MergedKeyBlocks>>> sortedKeyBlocks;
69
70 uint32_t numBytesPerTuple = UINT32_MAX; // encoding size
71 std::vector<StrKeyColInfo> strKeyColsInfo;
72};
