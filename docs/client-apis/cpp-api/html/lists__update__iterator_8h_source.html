8/*
9 * This class implements the CSR reconstruction algorithm to update lists transactionally.
10 * It is designed to be used as follows through calls to updateLists:
11 * auto updateItr = ListsUpdateIteratorFactory::getListsUpdateIterator(this);
12 * for (auto updatedChunkItr = listsUpdatesStore.updatedChunks.begin();
13 * updatedChunkItr != listsUpdatesStore.updatedChunks.end(); ++updatedChunkItr) {
14 * for (auto updatedNodeOffsetItr = updatedChunkItr->second->begin();
15 * updatedNodeOffsetItr != updatedChunkItr->second->end(); updatedNodeOffsetItr++) {
16 * updateItr.updateList(nodeOffset, inMemList);
17 * }
18 * }
19 * Two important requirements for correctly using this class are: (i) that the
20 * calls to updateList need to happen in ascending order of nodeOffsets; and (ii) there should not
21 * be multiple calls to writeInMemListToListPages for the same nodeOffset.
22 *
23 * The CSR re-construction algorithm is currently very basic and slides each small list to the right
24 * or left when updates to a chunk arrives (large lists don't need sliding because recall that they
25 * are not part of the CSR and they get their own list of pages), simply by reading them in their
26 * entirety and writing them into their new places. Small lists that become large, get their own
27 * page lists. Large lists do not become small if they become small. Large lists are also written in
28 * their entirety.
29 *
30 * We clarify the following abbreviations and conventions in the variables used in these files:
31 * <ul>
32 * <li> idxInPageList: refers to an index in the page list of a chunk or large list and is
33 * consecutive from 0 to # pages in the page list.
34 * <li> listPageIdx: physical pageIdx in the .lists file.
35 * </ul>
36 */

96class AdjListsUpdateIterator : public ListsUpdateIterator {

100private:
101 inline void updateLargeListHeaderIfNecessary(uint32_t largeListIdx) override {
102 common::list_header_t newHeader = ListHeaders::getLargeListHeader(largeListIdx);
103 lists->getHeaders()->headersDiskArray->update(curUnprocessedNodeOffset, newHeader);
104 }
105 inline void updateSmallListHeaderIfNecessary(
106 common::list_header_t oldHeader, common::list_header_t newHeader) override {
107 if (newHeader != oldHeader) {
108 lists->getHeaders()->headersDiskArray->update(curUnprocessedNodeOffset, newHeader);
109 }
110 }
111 inline bool isLargeListAfterInsertion(
112 common::list_header_t oldHeader, uint64_t numElementsAfterInsertion) override {
113 return ListHeaders::isALargeList(oldHeader) ||
114 numElementsAfterInsertion * lists->elementSize > common::DEFAULT_PAGE_SIZE;
115 }
