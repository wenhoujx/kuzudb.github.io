11struct Mask {
12public:
13 Mask(uint64_t size, uint8_t maskedFlag) : maskedFlag{maskedFlag} {
14 data = std::make_unique<uint8_t[]>(size);
15 std::fill(data.get(), data.get() + size, 0);
16 }
17
18 // Notice: This function is not protected with a lock for concurrent writes because of the
19 // special use case that there is no mixed reads and writes to the mask, and all writes to the
20 // mask try to set a position to the same value, thus it doesn't matter which thread succeeds.
21 inline void setMask(uint64_t pos, uint8_t maskerIdx, uint8_t maskValue) {
22 // Note: blindly update mask does not parallel well, so we minimize write by first checking
23 // if the mask is true or not.
24 if (data[pos] == maskerIdx) {
25 data[pos] = maskValue;
26 }
27 }
28 inline bool isMasked(uint64_t pos) { return data[pos] == maskedFlag; }
29
30private:
31 // The value of maskedFlag is equivalent to the num of maskers passed. It is used to check if a
32 // value is selected by all maskers or not. Each masker will increment its selected value by 1.
33 uint8_t maskedFlag;
34 std::unique_ptr<uint8_t[]> data;
35};
